## commands
  ```
  mkdir shop-review
  cd shop-review
  ```
  ```
  npm install expo
  npm install expo-cli
  expo init shop-review-app (expo-template-blank-typescript)
  ```
  ```
  npm install expo-cli
  cd shop-review-app
  expo start
  ```
